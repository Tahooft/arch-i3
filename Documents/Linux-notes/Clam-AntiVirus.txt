pacman -S clamav
sudo mkdir  /var/lib/clamav
sudo chown -R th:th /var/lib/clamav

# Note:
# You will need to run freshclam before starting the service for the first time or you will run into trouble/errors which will prevent ClamAV from starting correctly.

freshclam
systemctl start clamav-freshclam.service
systemctl enable clamav-freshclam.service



clamscan --recursive --infected --max-filesize=4000M --max-scansize=4000M -l clamlog.txt /home/

clamscan --max-filesize=20000M --max-scansize=20000M -r --bell -v -i --move=/path/to/infected/files / --exclude-dir=/proc --exclude-dir=/sys --exclude-dir=/dev > results.txt

# Test
curl https://secure.eicar.org/eicar.com.txt | clamscan -

# Get all the virus signatures
 yay -S clamav-unofficial-sigs


# Run the following command to display which signatures are being loaded by clamav
clamscan --debug 2>&1 /dev/null | grep "loaded"
